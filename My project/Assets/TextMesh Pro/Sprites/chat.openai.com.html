    public bool Loading { get; set; }
    public Quiz[] Quizzes { get; set; }

    private void Start()
    {
        Quizzes = new Quiz[0];
        Loading = false;
        
        LoadQuizzesAsync(5); // Změňte požadovaný počet kvízů podle potřeby
    }

    public async void LoadQuizzesAsync(int num)
    {
        Loading = true;
        try
        {
            string url = $"http://localhost:5191/api/Quiz/random/{num}";
            using (HttpClient client = new HttpClient())
            {
                HttpResponseMessage response = await client.GetAsync(url);
                if (response.IsSuccessStatusCode)
                {
                    string jsonResponse = await response.Content.ReadAsStringAsync();
                    Quizzes = JsonConvert.DeserializeObject<Quiz[]>(jsonResponse); 
                }
                else
                {
                    Debug.Log("Neúspěšný požadavek: " + response.StatusCode);
                }
            }
        }
        catch (HttpRequestException ex)
        {
            Debug.Log("Nepodařilo se připojit k serveru: " + ex.Message);
            Loading = false;
        }
        catch (Exception ex)
        {
            Debug.Log("Neočekávaná chyba: " + ex.Message);
            Loading = false;
        }
        Loading = false;
    }
}

Pro aktualizovanou třídu Quiz a formátování hodnot v metodě ExtractValuesFromFormattedString můžete použít následující kód:

public static Quiz ExtractQuizFromFormattedString(string input)
{
    string pattern = @"maxP\. (\d+), currentP\. (\d+), R\.Status (\w+), (.+),\[(.+)\]";

    Match match = Regex.Match(input, pattern);
    if (match.Success)
    {
        int maxPlayerCount = int.Parse(match.Groups[1].Value);
        int currentPlayerCount = int.Parse(match.Groups[2].Value);
        Quiz.RoomState roomState = (Quiz.RoomState)Enum.Parse(typeof(Quiz.RoomState), match.Groups[3].Value);
        DateTime createdAt = DateTime.Parse(match.Groups[4].Value);
        int[] players = match.Groups[5].Value.Split(", ").Select(int.Parse).ToArray();

        Quiz quiz = new Quiz(maxPlayerCount, currentPlayerCount, roomState, createdAt);
        quiz.Players = players;
        return quiz;
    }
    else
    {
        Debug.Log("Chyba");
        return null;
    }
}

Tato metoda přijímá formátovaný řetězec a pomocí regulárního výrazu extrahuje hodnoty pro jednotlivé vlastnosti Quiz. Poté jsou hodnoty převedeny na odpovídající datové typy a použity k vytvoření nového objektu Quiz. Pole Players je naplněno s rozdělenými a převedenými hodnotami ze závorkového seznamu.
